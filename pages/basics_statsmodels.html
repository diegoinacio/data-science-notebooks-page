or

Datasets¶

Statsmodels provides list of available datasets samples that can be read as pandas dataframes for testing.

In [4]:
# Load dataset
df1 = sm.datasets.cancer.load_pandas().data

# Plot dataset
df1.plot.scatter(x="population", y="cancer", s=10)
plt.show()

Time series analysis¶

In [7]:
# Load dataset
df2 = sm.datasets.co2.load_pandas().data
df2 = df2[df2.index >= "1995"]

# Plot dataset
df2.plot()

# Plot average mean
(
    df2.co2
    .rolling(7, center=True)
    .mean()
    .plot(c="red", label="average mean")
)

plt.legend()
plt.show()
